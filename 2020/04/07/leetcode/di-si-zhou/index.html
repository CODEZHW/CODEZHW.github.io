private LinkedList<TreeNode> queue = new LinkedList<TreeNode>();

public int[] levelOrder(TreeNode root) {
    if(root ==null){
        return new int[]{};
    }
    int[] res = new int[1024];
    queue.add(root);
    int count = 0;
    while(!queue.isEmpty()){
        TreeNode node = queue.pop();
        res[count++] = node.val;
        if(node.left!=null){
            queue.add(node.left);
        }
        if(node.right != null){
            queue.add(node.right);
        }
    }
    return Arrays.copyOf(res, count);
}

public int maxArea(int[] height) {
        int i = 0;
        int j = height.length-1;
        int max = 0;
        while(i<j){
            int newmax = Math.min(height[j], height[i]) * (j-i);

                j--;
            }
        }
