public int countNegatives(int[][] grid) {
        int res = 0;

        for(int i=0; i<grid.length; i++){
            int left = 0;
            int right = grid[i].length;
            while(left < right){
                int mid = left + (right - left)/2;
                if(grid[i][mid] < 0){
                    if(mid==0){
                        res +=grid[i].length;
                        break;
                    }
                    if(grid[i][mid-1] >= 0){
                        res +=grid[i].length - mid;
                        break;
                    }else{
                        right = mid;
                    }
                }else{
                    left= mid+1;
                }
            }
        }
        return res;
    }

                    return false;

public int[] maxSlidingWindow(int[] nums, int k) {
        if(nums.length==0){
            return nums;
        }
        int left = 0;
        int [] res = new int [nums.length-(k-1)];
        while(left <= nums.length-k){
            int right = left + k;
            int max = Integer.MIN_VALUE;
            for(int i=left; i<right; i++ ){
                max = Math.max(nums[i], max);
            }
            res[left++] = max;
        }
        return res;
    }
