public static int longestPalindrome ( String s ) {

    int[] c = new int[26 + 26];
    for (char ss : s.toCharArray ( )) {
        if (ss >= 'a') {
            c[ss - 'a'] += 1;
        } else {
            c[ss - 'A' + 26] += 1;
        }
    }
    int res = 0;
    int odd = 0;
    for (int cc : c) {
        res += cc;
        if (cc % 2 == 1) {
            odd += 1;
        }
    }
    return odd == 0 ? res : res - odd + 1;
}
